    <script>

        /**

         * Hides the modal and displays the final flashing message.

         */

        window.showFinalMessage = function () {

            const modal = document.getElementById('surprise-modal');

            const finalMessage = document.getElementById('final-message');

            const finalMessageText = document.getElementById('final-text');


            // Hide the main surprise modal

            modal.classList.remove('opacity-100', 'visible');

            modal.classList.add('opacity-0', 'invisible');


            // Show the final message overlay after the modal starts hiding

            setTimeout(() => {

                finalMessage.classList.remove('opacity-0', 'invisible');

                finalMessage.classList.add('opacity-100', 'visible');

                

                // Apply the flashing effect to the text

                finalMessageText.classList.add('flashing');

            }, 500); 

        }


        /**

         * Fades out the message card and fades in the photo section.

         */

        window.showPhoto = function () {

            const messageCard = document.getElementById('message-card');

            const photoSection = document.getElementById('photo-section');


            // Start fade out/scale down of message card

            messageCard.classList.add('opacity-0', 'scale-95');


            setTimeout(() => {

                // Ensure photo section is positioned absolutely within the wrapper

                photoSection.classList.remove('opacity-0', 'invisible', 'scale-95');

                photoSection.classList.add('opacity-100', 'scale-100');

            }, 500); // Wait for message card to start fading

        }


        /**

         * Shows the anniversary modal after a delay for the surprise effect.

         */

        window.onload = () => {

            const modal = document.getElementById('surprise-modal');

            const preLoadMessage = document.getElementById('pre-load-message');


            // Hide the initial message gracefully

            preLoadMessage.classList.add('opacity-0');

            

            // Wait 1.5 seconds for dramatic effect, then show the surprise

            setTimeout(() => {

                // Completely remove the initial message from display

                preLoadMessage.style.display = 'none';


                // Make the modal visible and opaque

                modal.classList.remove('opacity-0', 'invisible');

                modal.classList.add('opacity-100', 'visible');

            }, 1500); // 1.5 second delay

        };
